// so the website sidebar can show what people write inside Telegram.

// Send me your current app/api/telegram/route.ts if you want me to merge it safely.

async function saveCommunityTelegramMessage(update: any) {
  const message = update?.message || update?.edited_message;

  if (!message?.chat?.id || !message?.text) return;

  const groupId = String(process.env.TELEGRAM_GROUP_ID || "");

  if (!groupId || String(message.chat.id) !== groupId) return;
  if (message.from?.is_bot) return;

  const senderName =
    message.from?.username
      ? `@${message.from.username}`
      : [message.from?.first_name, message.from?.last_name].filter(Boolean).join(" ") ||
        "Telegram member";

  await supabaseFetch("broke_community_messages?on_conflict=telegram_message_id", {
    method: "POST",
    headers: {
      Prefer: "resolution=ignore-duplicates",
    },
    body: JSON.stringify({
      sender_name: senderName,
      username: message.from?.username ?? null,
      text: String(message.text).slice(0, 500),
      source: "telegram",
      telegram_message_id: message.message_id,
      created_at: new Date((message.date || Math.floor(Date.now() / 1000)) * 1000).toISOString(),
    }),
  });
}
